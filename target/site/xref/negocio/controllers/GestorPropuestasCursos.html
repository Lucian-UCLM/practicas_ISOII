15  	public CursoPropio realizarPropuestaCurso(String id, String nombre, int ECTS, Date fechaInicio, Date fechaFinal,
16  			double tasaMatricula, int edicion, EstadoCurso estado, TipoCurso tipo, String director, String secretario,
17  			int centro) {
18  		Centro centronew = new Centro(centro, "", "");
19  		ProfesorUCLM directornew = new ProfesorUCLM(director, CategoriaProfesor.ASOCIADO, centronew);
20  		ProfesorUCLM secretarionew = new ProfesorUCLM(secretario, CategoriaProfesor.ASOCIADO, centronew);
21  		CursoPropio curso = new CursoPropio(id, nombre, ECTS, fechaInicio, fechaFinal, tasaMatricula, edicion, estado,
22  				tipo, centronew, directornew, secretarionew);
23  		cursodao.crearNuevoCurso(curso);
24  		return curso;
25  	}
