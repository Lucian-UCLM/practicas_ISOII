133 		
134 		Object[][] matriz = new Object[numFilas][numColumnas];
135 		String[] campos = new String [numColumnas];
136 		System.arraycopy(arrayToCopy, 0, campos, 0, numColumnas);

138 		ingresosCantidadLabel.setText(String.valueOf(ingresos) + " €");
139 		ingresosCantidadLabel.setForeground(Color.BLUE);
140 		
141 		for (int i = 0; i < listaConsulta.size(); i++) {
142 			matriz[i][0] = listaConsulta.get(i).getNombre();
143 		}
144 
145 		for (int j = 0; j < campos.length; j++) {
146  
147 			if (campos[j].equals("Centro")) {
148 				for (int i = 0; i < listaConsulta.size(); i++) {
149 					matriz[i][j] = listaConsulta.get(i).getIdCentro();
150 				}
151 			}else if (campos[j].equals("Créditos")) {
152 				for (int i = 0; i < listaConsulta.size(); i++) {
153 					matriz[i][j] = listaConsulta.get(i).getECTS();
154 				}
155 			}else if (campos[j].equals("Importe")) {
156 				for (int i = 0; i < listaConsulta.size(); i++) {
157 					matriz[i][j] = listaConsulta.get(i).getTasaMatricula();
158 				}
159 			}else if (campos[j].equals("Edición")) {
160 				for (int i = 0; i < listaConsulta.size(); i++) {
161 					matriz[i][j] = listaConsulta.get(i).getEdicion();
162 				}
163 			}else if (campos[j].equals("Estado")) {
164 				for (int i = 0; i < listaConsulta.size(); i++) {
165 					matriz[i][j] = listaConsulta.get(i).getEstado();
