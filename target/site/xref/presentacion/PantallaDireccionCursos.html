362 			fechafin.setDate(curso.getFechaFin());
363

365 			creditostext.setText(curso.getECTS() + "");

369 			estadotext.setText(curso.getEstado().toString());

371 			ediciontext.setEditable(false);
372 			idtext.setEditable(false);
373 			edicionlabel.setEnabled(false);

377 			edicionbutton.setEnabled(true);
378 			for (int i = 0; i < TipoCurso.values().length; i++) {
379 				if (tipocombobox.getModel().getElementAt(i).toString().equals(curso.getTipo().toString())) {
380 					tipocombobox.setSelectedItem(tipocombobox.getModel().getElementAt(i));
381 				}
382 			}
383 			for (int i = 0; i < centrolist.getModel().getSize(); i++) {
384 				if (centrolist.getModel().getElementAt(i).toString().equals(curso.getIdCentro() + "")) {
385 					centrolist.setSelectedIndex(i);
386 				}
387 			}
388 			for (int i = 0; i < directorlist.getModel().getSize(); i++) {
389 				if (directorlist.getModel().getElementAt(i).equals(curso.getIdDirector())) {
390 					directorlist.setSelectedIndex(i);
391 				}
392 			}
393 			for (int i = 0; i < secretariolist.getModel().getSize(); i++) {
394 				if (secretariolist.getModel().getElementAt(i).equals(curso.getIdSecretario())) {
395 					secretariolist.setSelectedIndex(i);
396 				}
397 			}

399 		}
